# 局部

在当前文件夹立创建constants.js文件

```
//  constants.js

const list = [
    {a: 1},
    {b: 2}
];

export {
  list, // 列表过滤数据结构
}

// 使用：import { list } from "./constants";
```

# 全局

在common文件夹下创建constants文件夹

```
//  list.js

const list = [
    {a: 1},
    {b: 2}
];

export {
  list, // 列表过滤数据结构
}


```

common/constants/index.js 该文件作为转发聚合

```
// index.js

export { list } from './list'; // 首页 -- 列表管理 -- 列表表头

// 使用：import { list } from "common/constants/index"; // 需要配合路径别名
```
